24.打印乘法口诀表


#include <stdio.h>
int main() {
    for (int i = 1; i <= 9; i++) {
        for (int j = 1; j <= i; j++)
            printf("\t%d*%d=%d", i, j, i * j);
        printf("\n");
    }
    return 0;
}

        int i, x;

        for (i = 2; i <= sqrt(x); i++)
            if (x % i == 0)
                break;
        if (i > sqrt(x))
            printf("%d is a prime number.\n", x);
        else
            printf("%d is not a prime number.\n", x);
