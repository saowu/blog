34.输入两个整数m和n，求其最大公约数和最小公倍数


#include <stdio.h>
#include <math.h>

int main() {
    int m, n, r, p;
    printf("input m,n: ");
    scanf("%d,%d", &m, &n);
    p = m * n;
    do {
        r = m % n;
        m = n;
        n = r;
    } while (r != 0);
    printf("最大公约数:%d\n", m);
    printf("最小公倍数:%d", p / m);
    return 0;
}

40.如果一个数恰好等于它的因子之和，这个数为“完数”（例如:6=1+2+3），找出1000以内的完数，按照:“6 its factors are 1 2 3” 格式打印


#include <stdio.h>
#include <math.h>
int main() {
    int s;
    for (int i = 1; i <= 1000; i++) {
        s = 0;
        for (int j = 1; j <= i / 2; j++)
            if (i % j == 0)
                s += j;
        if (s == i) {
            printf("%d it's factors are ", i);
            for (int j = 1; j <= i / 2; j++) {
                if (i % j == 0)
                    printf("%d ", j);
            }
            printf("\n");
        }
    }
    return 0;
}

44.判断一个数是否为素数


#include <stdio.h>
#include <math.h>

int main() {
    int m, i;
    scanf("%d", &m);
    for (i = 2; i <= m / 2; i++)
        if (m % i == 0)
            break;
    if (m/2 == i-1)
        printf("Yes");
